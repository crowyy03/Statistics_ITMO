confidence_levels = [0.90, 0.95, 0.99]
alphas = [0.10, 0.05, 0.01]

def calculate_percentile_ci(bootstrap_estimates, alpha):
    lower_percentile = 100 * (alpha / 2)
    upper_percentile = 100 * (1 - alpha / 2)
    ci_lower = np.percentile(bootstrap_estimates, lower_percentile)
    ci_upper = np.percentile(bootstrap_estimates, upper_percentile)
    return ci_lower, ci_upper

print("\n--- 3. Построение доверительных интервалов (Процентильный метод) ---")

cis_mean = {}
cis_median = {}

for level, alpha_val in zip(confidence_levels, alphas):

    ci_median_lower, ci_median_upper = calculate_percentile_ci(bootstrap_medians, alpha_val)

mu_true_coverage = 0
sigma_true_coverage = 1

N_values_coverage = [50, 100, 200, 500, 1000]
B_values_coverage = [100, 200, 400, 1600, 3200]
num_simulations = 100

coverage_results = np.zeros((len(N_values_coverage), len(B_values_coverage)))
coverage_table_data = []

print("\n--- 5. Проверка покрытия интервалов (для N(0,1)) ---")

for i, n in enumerate(N_values_coverage):
    for j, b in enumerate(B_values_coverage):
        contained_count = 0
        for _ in range(num_simulations):
            sample_coverage = np.random.normal(loc=mu_true_coverage, scale=sigma_true_coverage, size=n)
            bootstrap_means_coverage = []
            for _ in range(b):
                bootstrap_sample_coverage = np.random.choice(sample_coverage, size=n, replace=True)
                bootstrap_means_coverage.append(np.mean(bootstrap_sample_coverage))

            ci_lower, ci_upper = calculate_percentile_ci(bootstrap_means_coverage, alpha=0.05)

            if ci_lower <= mu_true_coverage <= ci_upper:
                contained_count += 1

        coverage_proportion = contained_count / num_simulations
        coverage_results[i, j] = coverage_proportion

        if b == 1000:
            coverage_table_data.append({'Объем выборки N': n, 'Число итераций B': b, 'Доля покрытия': f'{coverage_proportion:.2f}'})

        print(f"N={n}, B={b}: Доля покрытия = {coverage_proportion:.2f}")
